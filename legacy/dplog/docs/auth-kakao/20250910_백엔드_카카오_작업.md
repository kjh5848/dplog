const crypto = require('crypto');

// 카카오 토큰 암호화
function encryptKakaoToken(token) {
  const cipher = crypto.createCipher('aes-256-cbc', process.env.KAKAO_TOKEN_ENCRYPTION_KEY);
  let encrypted = cipher.update(token, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  return encrypted;
}

// 카카오 토큰 복호화
function decryptKakaoToken(encryptedToken) {
  const decipher = crypto.createDecipher('aes-256-cbc', process.env.KAKAO_TOKEN_ENCRYPTION_KEY);
  let decrypted = decipher.update(encryptedToken, 'hex', 'utf8');
  decrypted += decipher.final('utf8');
  return decrypted;
}

## 🌐 카카오 API 연동

### 1. 토큰 발급 API

```javascript
// 카카오 토큰 발급
async function getKakaoToken(authCode) {
  const response = await fetch('https://kauth.kakao.com/oauth/token', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/x-www-form-urlencoded',
    },
    body: new URLSearchParams({
      grant_type: 'authorization_code',
      client_id: process.env.KAKAO_REST_API_KEY,
      redirect_uri: process.env.KAKAO_REDIRECT_URI,
      code: authCode,
    }),
  });

  if (!response.ok) {
    throw new Error('카카오 토큰 발급 실패');
  }

  return response.json();
}
```

## 🔄 비즈니스 로직 구현

### 1. 사용자 매칭 로직

```javascript
async function findOrCreateUser(kakaoUserInfo) {
  const { id: kakaoId, kakao_account } = kakaoUserInfo;
  const email = kakao_account?.email;
  const nickname = kakao_account?.profile?.nickname || kakaoUserInfo.properties?.nickname;
  const profileImage = kakao_account?.profile?.profile_image_url || kakaoUserInfo.properties?.profile_image;

  // 1. 카카오 ID로 기존 사용자 찾기
  let user = await findUserByKakaoId(kakaoId);
  
  if (user) {
    // 기존 카카오 사용자 - 정보 업데이트
    await updateKakaoUserInfo(user.id, {
      kakao_nickname: nickname,
      kakao_profile_image: profileImage,
      kakao_email: email,
    });
    return user;
  }

  // 2. 이메일로 기존 사용자 찾기 (계정 연동 케이스)
  if (email) {
    user = await findUserByEmail(email);
    if (user) {
      // 기존 일반 사용자에 카카오 정보 연동
      await linkKakaoToUser(user.id, {
        kakao_id: kakaoId,
        kakao_nickname: nickname,
        kakao_profile_image: profileImage,
        kakao_email: email,
        kakao_linked_at: new Date(),
      });
      return user;
    }
  }

  // 3. 신규 사용자 생성
  const newUser = await createKakaoUser({
    username: `kakao_${kakaoId}`,
    userName: nickname || `카카오사용자${kakaoId.slice(-4)}`,
    email: email,
    kakao_id: kakaoId,
    kakao_nickname: nickname,
    kakao_profile_image: profileImage,
    kakao_email: email,
    kakao_linked_at: new Date(),
  });

  return newUser;
}
```

### 2. JWT 토큰 생성

```javascript
function generateTokens(user) {
  const payload = {
    userId: user.id,
    username: user.username,
    email: user.email,
    isKakaoUser: !!user.kakao_id,
  };

  const accessToken = jwt.sign(payload, process.env.JWT_SECRET, {
    expiresIn: process.env.JWT_EXPIRES_IN,
  });

  const refreshToken = jwt.sign(
    { userId: user.id },
    process.env.JWT_REFRESH_SECRET,
    { expiresIn: process.env.JWT_REFRESH_EXPIRES_IN }
  );

  return { accessToken, refreshToken };
}
```
